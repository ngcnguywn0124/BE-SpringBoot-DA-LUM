-- Bang truong dai hoc
CREATE TABLE universities (
    university_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    university_name VARCHAR(255) NOT NULL UNIQUE,
    short_name VARCHAR(50),
    slug VARCHAR(255) UNIQUE,
    city VARCHAR(100),
    address TEXT,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- Bang co so / campus cua truong
CREATE TABLE campuses (
    campus_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    university_id UUID NOT NULL REFERENCES universities(university_id) ON DELETE CASCADE,
    campus_name VARCHAR(255) NOT NULL,
    slug VARCHAR(255) UNIQUE,
    address TEXT,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- Bang nguoi dung
CREATE TABLE users (
    user_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    email VARCHAR(255) UNIQUE NOT NULL,
    password_hash VARCHAR(255),
    phone_number VARCHAR(20),
    full_name VARCHAR(255) NOT NULL,
    avatar_url TEXT,
    avatar_id VARCHAR(255),
    cover_url TEXT,
    cover_id VARCHAR(255),
    date_of_birth DATE,
    gender VARCHAR(10) CHECK (gender IN ('male', 'female', 'other')),

    -- Thong tin sinh vien
    student_id VARCHAR(50),
    university_id UUID REFERENCES universities(university_id),
    campus_id UUID REFERENCES campuses(campus_id),
    faculty VARCHAR(255),
    graduation_year INTEGER,

    -- Ho so ca nhan
    bio TEXT,
    location VARCHAR(255),

    -- Thong ke hoat dong
    reputation_score DECIMAL(3,2) DEFAULT 0.00,
    total_sales INTEGER DEFAULT 0,
    total_purchases INTEGER DEFAULT 0,
    followers_count INTEGER DEFAULT 0,
    following_count INTEGER DEFAULT 0,
    response_rate DECIMAL(5,2) DEFAULT 0.00,
    response_time VARCHAR(50),

    -- Trang thai tai khoan
    is_email_verified BOOLEAN DEFAULT FALSE,
    is_phone_verified BOOLEAN DEFAULT FALSE,
    is_student_verified BOOLEAN DEFAULT FALSE,
    is_active BOOLEAN DEFAULT TRUE,
    failed_login_attempts INTEGER DEFAULT 0,
    locked_until TIMESTAMP WITH TIME ZONE,


    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    last_login_at TIMESTAMP WITH TIME ZONE
    last_seen_at updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
);

-- Bang dang nhap bang mang xa hoi (OAuth)
CREATE TABLE oauth_accounts (
    oauth_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID NOT NULL REFERENCES users(user_id) ON DELETE CASCADE,
    provider VARCHAR(50) NOT NULL CHECK (provider IN ('google', 'facebook')),
    provider_user_id VARCHAR(255) NOT NULL,
    provider_email VARCHAR(255),
    access_token TEXT,
    refresh_token TEXT,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    UNIQUE(provider, provider_user_id)
);

CREATE TABLE IF NOT EXISTS roles
(
    id          UUID        PRIMARY KEY DEFAULT gen_random_uuid(),
    name        VARCHAR(50) NOT NULL UNIQUE,
    slug        VARCHAR(100) UNIQUE,
    description VARCHAR(255),
    created_at  TIMESTAMP   NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at  TIMESTAMP   NOT NULL DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT chk_role_name CHECK (name IN ('ROLE_USER', 'ROLE_ADMIN', 'ROLE_MODERATOR', 'ROLE_SUPER_ADMIN'))
);

CREATE TABLE IF NOT EXISTS permissions
(
    id          UUID         PRIMARY KEY DEFAULT gen_random_uuid(),
    name        VARCHAR(100) NOT NULL UNIQUE,
    slug        VARCHAR(150) UNIQUE,
    description VARCHAR(255),
    resource    VARCHAR(50)  NOT NULL,
    action      VARCHAR(50)  NOT NULL,
    created_at  TIMESTAMP    NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at  TIMESTAMP    NOT NULL DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE IF NOT EXISTS user_roles
(
    user_id     UUID      NOT NULL, -- UUID khop voi users.user_id
    role_id     UUID      NOT NULL,
    assigned_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (user_id, role_id),
    CONSTRAINT fk_user_roles_user FOREIGN KEY (user_id) REFERENCES users (user_id) ON DELETE CASCADE,
    CONSTRAINT fk_user_roles_role FOREIGN KEY (role_id) REFERENCES roles (id) ON DELETE CASCADE
);

CREATE TABLE IF NOT EXISTS role_permissions
(
    role_id       UUID      NOT NULL,
    permission_id UUID      NOT NULL,
    assigned_at   TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (role_id, permission_id),
    CONSTRAINT fk_role_permissions_role FOREIGN KEY (role_id) REFERENCES roles (id) ON DELETE CASCADE,
    CONSTRAINT fk_role_permissions_permission FOREIGN KEY (permission_id) REFERENCES permissions (id) ON DELETE CASCADE
);

CREATE INDEX idx_user_roles_user_id ON user_roles (user_id);
CREATE INDEX idx_user_roles_role_id ON user_roles (role_id);
CREATE INDEX idx_role_permissions_role_id ON role_permissions (role_id);
CREATE INDEX idx_role_permissions_permission_id ON role_permissions (permission_id);
CREATE INDEX idx_permissions_resource_action ON permissions (resource, action);

CREATE TABLE categories (
    category_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    category_name VARCHAR(255) NOT NULL,
    parent_category_id UUID REFERENCES categories(category_id),
    slug VARCHAR(255) UNIQUE,

    image_url TEXT,

    updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP

-- Ap dung trigger updated_at cho categories
CREATE TRIGGER update_categories_updated_at
    BEFORE UPDATE ON categories
    FOR EACH ROW EXECUTE FUNCTION update_updated_at_column();

-- Bang san pham / tin dang
-- Ghi chu: listing_type = muc dich dang (ban/doi); transaction_type = hinh thuc giao nhan
CREATE TABLE products (
    product_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    seller_id UUID NOT NULL REFERENCES users(user_id) ON DELETE CASCADE,
    category_id UUID NOT NULL REFERENCES categories(category_id),

    -- Thong tin co ban
    title VARCHAR(120) NOT NULL,
    description TEXT NOT NULL,
    slug VARCHAR(255) UNIQUE,
    condition VARCHAR(20) NOT NULL CHECK (condition IN ('new', 'like_new', 'used', 'old', 'broken')),
    -- new      : Moi 100%, chua boc hop
    -- like_new : Nhu moi 99%, cuc dep
    -- used     : Da qua su dung, binh thuong
    -- old      : Cu / van dung tot, co tray xuoc
    -- broken   : Hong / lay linh kien

    -- Gia
    price DECIMAL(12,2),
    is_free BOOLEAN DEFAULT FALSE,
    is_negotiable BOOLEAN DEFAULT TRUE,

    -- Hinh thuc dang tin (muc dich)
    listing_type VARCHAR(20) NOT NULL DEFAULT 'sell' CHECK (listing_type IN ('sell', 'exchange', 'both')),
    exchange_preferences TEXT,

    -- Hinh thuc giao nhan
    transaction_type VARCHAR(20) NOT NULL DEFAULT 'meetup' CHECK (transaction_type IN ('meetup', 'delivery', 'both')),
    meeting_point TEXT,

    -- Vi tri (theo truong & co so)
    university_id UUID REFERENCES universities(university_id),
    campus_id UUID REFERENCES campuses(campus_id),

    -- Thong tin lien he bo sung
    contact_name VARCHAR(255),
    contact_phone VARCHAR(20),
    zalo_link TEXT,
    facebook_link TEXT,

    -- Trang thai
    status VARCHAR(20) DEFAULT 'available' CHECK (status IN (
        'available',
        'pending',
        'hidden',
        'expired',
        'sold',
        'deleted',
        'admin_hidden'
    )),
    previous_status VARCHAR(20),

    -- Thoi han & gia han
    expiry_days INTEGER DEFAULT 30,
    renewal_count INTEGER DEFAULT 0,
    expires_at TIMESTAMP WITH TIME ZONE,

    -- Thong ke
    view_count INTEGER DEFAULT 0,
    favorite_count INTEGER DEFAULT 0,
    message_count INTEGER DEFAULT 0,

    is_featured BOOLEAN DEFAULT FALSE,

    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    approved_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    sold_at TIMESTAMP WITH TIME ZONE
);

-- Bang hinh anh san pham
CREATE TABLE product_images (
    image_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    product_id UUID NOT NULL REFERENCES products(product_id) ON DELETE CASCADE,
    image_url TEXT NOT NULL,
    image_cloud_id VARCHAR(255),
    display_order INTEGER DEFAULT 0,
    is_primary BOOLEAN DEFAULT FALSE,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);


-- Bang dinh nghia thuoc tinh dong theo danh muc (danh cho admin cau hinh)
CREATE TABLE product_attributes (
    attribute_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    category_id UUID NOT NULL REFERENCES categories(category_id),
    attribute_name VARCHAR(100) NOT NULL,
    attribute_type VARCHAR(20) CHECK (attribute_type IN ('text', 'number', 'boolean', 'select')),
    is_required BOOLEAN DEFAULT FALSE,
    options JSONB,
    display_order INTEGER DEFAULT 0
);

-- Bang gia tri thuoc tinh dong cua san pham
CREATE TABLE product_attribute_values (
    value_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    product_id UUID NOT NULL REFERENCES products(product_id) ON DELETE CASCADE,
    attribute_id UUID NOT NULL REFERENCES product_attributes(attribute_id),
    value TEXT NOT NULL,
    UNIQUE(product_id, attribute_id)
);

-- Bang tags / tu khoa
CREATE TABLE tags (
    tag_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    tag_name VARCHAR(100) NOT NULL UNIQUE,
    slug VARCHAR(150) UNIQUE,
    usage_count INTEGER DEFAULT 0,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP

-- Bang lien ket san pham va tags
CREATE TABLE product_tags (
    product_tag_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),

    tag_id UUID NOT NULL REFERENCES tags(tag_id) ON DELETE CASCADE,

-- Bang yeu thich / tin da luu
CREATE TABLE favorites (
    favorite_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID NOT NULL REFERENCES users(user_id) ON DELETE CASCADE,
    product_id UUID NOT NULL REFERENCES products(product_id) ON DELETE CASCADE,

    UNIQUE(user_id, product_id)
);


-- ============================================================
-- PHAN 3: GIAO DICH
-- ============================================================

-- Bang giao dich
-- Vong doi trang thai:
--   buyer_requested -> seller_confirmed -> meetup_confirmed -> payment_pending -> completed
--                                                                              -> cancelled (bat ky giai doan)
--                                                                              -> disputed
CREATE TABLE transactions (
    transaction_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    product_id UUID NOT NULL REFERENCES products(product_id),
    buyer_id UUID NOT NULL REFERENCES users(user_id),
    seller_id UUID NOT NULL REFERENCES users(user_id),

    transaction_type VARCHAR(20) NOT NULL CHECK (transaction_type IN ('sale', 'exchange')),

    status VARCHAR(30) DEFAULT 'buyer_requested' CHECK (status IN (
        'buyer_requested',
        'seller_confirmed',
        'meetup_confirmed',
        'payment_pending',
        'completed',
        'cancelled',
        'disputed'
    )),

    -- Thong tin gia da thoa thuan
    agreed_price DECIMAL(12,2),
    payment_method VARCHAR(30) CHECK (payment_method IN ('cash', 'transfer')),

    -- Thong tin gap mat
    meetup_location TEXT,
    meetup_time TIMESTAMP WITH TIME ZONE,

    -- Xac nhan song phuong - gap mat
    buyer_confirmed_meetup BOOLEAN DEFAULT FALSE,
    seller_confirmed_meetup BOOLEAN DEFAULT FALSE,

    -- Xac nhan song phuong - thanh toan
    buyer_confirmed_payment BOOLEAN DEFAULT FALSE,
    seller_confirmed_payment BOOLEAN DEFAULT FALSE,

    -- Hinh thuc giao nhan
    shipping_method VARCHAR(30) CHECK (shipping_method IN ('meetup', 'delivery')),
    shipping_address TEXT,
    shipping_fee DECIMAL(10,2),

    -- Trao doi (exchange)
    exchange_product_id UUID REFERENCES products(product_id),
    additional_payment DECIMAL(12,2),

    -- Huy / tranh chap
    cancellation_reason TEXT,
    cancelled_by UUID REFERENCES users(user_id),

    notes TEXT,

    -- Timeline
    requested_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    seller_confirmed_at TIMESTAMP WITH TIME ZONE,
    meetup_confirmed_at TIMESTAMP WITH TIME ZONE,
    completed_at TIMESTAMP WITH TIME ZONE,
    cancelled_at TIMESTAMP WITH TIME ZONE,

    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- Bang lich su trang thai giao dich
CREATE TABLE transaction_status_history (
    history_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    transaction_id UUID NOT NULL REFERENCES transactions(transaction_id) ON DELETE CASCADE,
    status VARCHAR(30) NOT NULL,
    changed_by UUID REFERENCES users(user_id),
    notes TEXT,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- PHAN 4: TIN NHAN & HOI THOAI
-- ============================================================

-- Bang cuoc hoi thoai
CREATE TABLE conversations (
    conversation_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    product_id UUID REFERENCES products(product_id) ON DELETE SET NULL,
    transaction_id UUID REFERENCES transactions(transaction_id) ON DELETE SET NULL,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- Bang nguoi tham gia hoi thoai
CREATE TABLE conversation_participants (
    participant_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    conversation_id UUID NOT NULL REFERENCES conversations(conversation_id) ON DELETE CASCADE,
    user_id UUID NOT NULL REFERENCES users(user_id) ON DELETE CASCADE,
    is_pinned BOOLEAN DEFAULT FALSE,
    last_read_at TIMESTAMP WITH TIME ZONE,
    is_active BOOLEAN DEFAULT TRUE,
    joined_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    UNIQUE(conversation_id, user_id)
);

-- Bang tin nhan
CREATE TABLE messages (
    message_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    conversation_id UUID NOT NULL REFERENCES conversations(conversation_id) ON DELETE CASCADE,
    sender_id UUID NOT NULL REFERENCES users(user_id) ON DELETE CASCADE,

    message_type VARCHAR(30) DEFAULT 'text' CHECK (message_type IN (
        'text',
        'image',
        'offer',
        'system',
        'transaction_event'
    )),

    content TEXT,
    attachment_url TEXT,
    offer_amount DECIMAL(12,2),

    -- Su kien giao dich (dung khi message_type = 'transaction_event')
    transaction_event_type VARCHAR(50) CHECK (transaction_event_type IN (
        'buyer_requested',
        'seller_confirmed',
        'meetup_confirmed',
        'payment_confirmed',
        'completed',
        'cancelled'
    )),

    -- Trang thai gui / nhan
    delivery_status VARCHAR(20) DEFAULT 'sent' CHECK (delivery_status IN (
        'sent',
        'delivered',
        'seen'
    )),

    is_edited BOOLEAN DEFAULT FALSE,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- Bang cai dat thong bao nguoi dung
CREATE TABLE notification_settings (
    setting_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID NOT NULL REFERENCES users(user_id) ON DELETE CASCADE UNIQUE,
    email_notifications BOOLEAN DEFAULT TRUE,
    push_notifications BOOLEAN DEFAULT TRUE,
    sms_notifications BOOLEAN DEFAULT FALSE,
    notification_preferences JSONB DEFAULT '{
        "new_message": true,
        "new_offer": true,
        "transaction_update": true,
        "product_sold": true,
        "product_expired": true,
        "review_received": true,
        "wishlist_update": true,
        "new_follower": true
    }'::jsonb,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- Bang bao cao vi pham
CREATE TABLE reports (
    report_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    reporter_id UUID NOT NULL REFERENCES users(user_id),
    reported_entity_type VARCHAR(20) NOT NULL CHECK (reported_entity_type IN (
        'product', 'user', 'review', 'message'
    )),
    reported_entity_id UUID NOT NULL,
    reason VARCHAR(50) NOT NULL CHECK (reason IN (
        'spam', 'fraud', 'inappropriate', 'fake', 'harassment', 'other'
    )),
    description TEXT NOT NULL,
    status VARCHAR(20) DEFAULT 'pending' CHECK (status IN (
        'pending', 'reviewing', 'resolved', 'rejected'
    )),
    admin_notes TEXT,
    resolved_by UUID REFERENCES users(user_id),
    resolved_at TIMESTAMP WITH TIME ZONE,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- Bang hanh dong quan tri
CREATE TABLE admin_actions (
    action_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    admin_id UUID NOT NULL REFERENCES users(user_id),
    action_type VARCHAR(50) NOT NULL CHECK (action_type IN (
        'ban_user', 'unban_user', 'delete_product', 'hide_product',
        'delete_review', 'resolve_report', 'feature_product'
    )),
    target_entity_type VARCHAR(20) NOT NULL,
    target_entity_id UUID NOT NULL,
    reason TEXT,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- Function: Cap nhat updated_at tu dong
CREATE OR REPLACE FUNCTION update_updated_at_column()
RETURNS TRIGGER AS $$
BEGIN
    NEW.updated_at = CURRENT_TIMESTAMP;
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- Ap dung trigger updated_at cho cac bang can thiet
CREATE TRIGGER update_users_updated_at
    BEFORE UPDATE ON users
    FOR EACH ROW EXECUTE FUNCTION update_updated_at_column();

CREATE TRIGGER update_products_updated_at
    BEFORE UPDATE ON products
    FOR EACH ROW EXECUTE FUNCTION update_updated_at_column();

CREATE TRIGGER update_transactions_updated_at
    BEFORE UPDATE ON transactions
    FOR EACH ROW EXECUTE FUNCTION update_updated_at_column();

CREATE TRIGGER update_messages_updated_at
    BEFORE UPDATE ON messages
    FOR EACH ROW EXECUTE FUNCTION update_updated_at_column();

CREATE TRIGGER update_conversations_updated_at
    BEFORE UPDATE ON conversations
    FOR EACH ROW EXECUTE FUNCTION update_updated_at_column();

-- Function: Cap nhat so luong yeu thich
CREATE OR REPLACE FUNCTION update_product_favorite_count()
RETURNS TRIGGER AS $$
BEGIN
    IF TG_OP = 'INSERT' THEN
        UPDATE products SET favorite_count = favorite_count + 1
        WHERE product_id = NEW.product_id;
    ELSIF TG_OP = 'DELETE' THEN
        UPDATE products SET favorite_count = GREATEST(favorite_count - 1, 0)
        WHERE product_id = OLD.product_id;
    END IF;
    RETURN NULL;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER update_favorite_count
    AFTER INSERT OR DELETE ON favorites
    FOR EACH ROW EXECUTE FUNCTION update_product_favorite_count();

-- Function: Cap nhat reputation score tu reviews
CREATE OR REPLACE FUNCTION update_user_reputation()
RETURNS TRIGGER AS $$
BEGIN
    UPDATE users SET reputation_score = (
        SELECT COALESCE(ROUND(AVG(rating)::NUMERIC, 2), 0.00)
        FROM reviews
        WHERE reviewee_id = NEW.reviewee_id
        AND is_visible = TRUE
    ) WHERE user_id = NEW.reviewee_id;
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER update_reputation
    AFTER INSERT OR UPDATE ON reviews
    FOR EACH ROW EXECUTE FUNCTION update_user_reputation();

-- Function: Ghi log lich su trang thai giao dich
CREATE OR REPLACE FUNCTION log_transaction_status_change()
RETURNS TRIGGER AS $$
BEGIN
    IF NEW.status <> OLD.status THEN
        INSERT INTO transaction_status_history (transaction_id, status, notes)
        VALUES (NEW.transaction_id, NEW.status, 'Status changed automatically');
    END IF;
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER log_status_change
    AFTER UPDATE ON transactions
    FOR EACH ROW EXECUTE FUNCTION log_transaction_status_change();

-- Function: Cap nhat last_read khi nguoi dung gui tin nhan
CREATE OR REPLACE FUNCTION update_sender_last_read()
RETURNS TRIGGER AS $$
BEGIN
    UPDATE conversation_participants
    SET last_read_at = CURRENT_TIMESTAMP
    WHERE conversation_id = NEW.conversation_id
    AND user_id = NEW.sender_id;
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER update_last_read_on_send
    AFTER INSERT ON messages
    FOR EACH ROW EXECUTE FUNCTION update_sender_last_read();

-- Function: Cap nhat followers_count khi follow/unfollow
CREATE OR REPLACE FUNCTION update_follower_counts()
RETURNS TRIGGER AS $$
BEGIN
    IF TG_OP = 'INSERT' THEN
        UPDATE users SET following_count = following_count + 1 WHERE user_id = NEW.follower_id;
        UPDATE users SET followers_count = followers_count + 1 WHERE user_id = NEW.following_id;
    ELSIF TG_OP = 'DELETE' THEN
        UPDATE users SET following_count = GREATEST(following_count - 1, 0) WHERE user_id = OLD.follower_id;
        UPDATE users SET followers_count = GREATEST(followers_count - 1, 0) WHERE user_id = OLD.following_id;
    END IF;
    RETURN NULL;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER update_follower_counts_trigger
    AFTER INSERT OR DELETE ON user_followers
    FOR EACH ROW EXECUTE FUNCTION update_follower_counts();

-- Function: Tu dong set expires_at khi tao san pham
CREATE OR REPLACE FUNCTION set_product_expiry()
RETURNS TRIGGER AS $$
BEGIN
    IF NEW.expires_at IS NULL THEN
        NEW.expires_at = CURRENT_TIMESTAMP + (COALESCE(NEW.expiry_days, 30) || ' days')::INTERVAL;
    END IF;
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER set_product_expiry_trigger
    BEFORE INSERT ON products
    FOR EACH ROW EXECUTE FUNCTION set_product_expiry();

-- View: Thong ke nguoi dung
CREATE MATERIALIZED VIEW user_statistics AS
SELECT
    u.user_id,
    u.full_name,
    u.avatar_url,
    u.reputation_score,
    u.followers_count,
    COUNT(DISTINCT p.product_id)                                            AS total_products_listed,
    COUNT(DISTINCT CASE WHEN p.status = 'sold' THEN p.product_id END)      AS total_products_sold,
    COUNT(DISTINCT tb.transaction_id)                                       AS total_purchases,
    COUNT(DISTINCT ts.transaction_id)                                       AS total_sales,
    COUNT(DISTINCT r.review_id)                                             AS total_reviews_received,
    COALESCE(ROUND(AVG(r.rating)::NUMERIC, 2), 0.00)                       AS average_rating
FROM users u
LEFT JOIN products p ON u.user_id = p.seller_id AND p.status <> 'deleted'
LEFT JOIN transactions tb ON u.user_id = tb.buyer_id AND tb.status = 'completed'
LEFT JOIN transactions ts ON u.user_id = ts.seller_id AND ts.status = 'completed'
LEFT JOIN reviews r ON u.user_id = r.reviewee_id AND r.is_visible = TRUE
GROUP BY u.user_id, u.full_name, u.avatar_url, u.reputation_score, u.followers_count;

CREATE UNIQUE INDEX idx_user_stats_user_id ON user_statistics(user_id);

-- View: San pham dang trending (nhieu luot xem, yeu thich gan day)
CREATE MATERIALIZED VIEW trending_products AS
SELECT
    p.product_id,
    p.title,
    p.price,
    p.is_free,
    p.condition,
    p.view_count,
    p.favorite_count,
    p.created_at,
    c.category_name,
    p.university_id,
    un.short_name                                   AS university_short_name,
    us.full_name                                    AS seller_name,
    us.reputation_score                             AS seller_reputation,
    (p.view_count * 0.3 + p.favorite_count * 0.7)  AS trending_score
FROM products p
JOIN categories c ON p.category_id = c.category_id
JOIN users us ON p.seller_id = us.user_id
LEFT JOIN universities un ON p.university_id = un.university_id
WHERE p.status = 'available'
AND p.created_at > CURRENT_TIMESTAMP - INTERVAL '30 days'
ORDER BY trending_score DESC
LIMIT 100;

CREATE UNIQUE INDEX idx_trending_product_id ON trending_products(product_id);

    FOR SELECT
    USING (
        sender_id = current_setting('app.current_user_id')::uuid

            SELECT 1 FROM conversation_participants
            WHERE conversation_id = messages.conversation_id
            AND user_id = current_setting('app.current_user_id')::uuid
        )
    );
